 76    # Helper to sanitize user inputs to mitigate prompt injection
 77    def _sanitize_user_text(self, s: str) -> str:
 78        lines = s.split('\n')
 79        sanitized_lines = []
 80        for line in lines:
 81            trimmed = line.strip()
 82            lower_trimmed = trimmed.lower()
 83            if lower_trimmed.startswith("ignore") or lower_trimmed.startswith("system:") or lower_trimmed.startswith("user:"):
 84                continue
 85            # Strip backticks to avoid escaping markdown prompt boundaries
 86            line_clean = line.replace("`", "")
 87            sanitized_lines.append(line_clean)
 88        return '\n'.join(sanitized_lines)

 90    # Helper to extract the first domain-like token from identity text
 91    def _extract_domain(self, text: str) -> str:
 92        words = text.split()
 93        for w in words:
 94            w_clean = w.strip(".,;:()\"'!?")
 95            if "://" in w_clean:
 96                parts = w_clean.split("://")
 97                if len(parts) > 1:
 98                    return parts[1].split('/')[0]
 99            if "." in w_clean and len(w_clean) > 4 and not w_clean.startswith("@"):
100                parts = w_clean.split(".")
101                last_part = parts[-1].split('/')[0].lower()
102                if last_part.isalnum() and 2 <= len(last_part) <= 6:
103                    res = w_clean.replace("https://", "").replace("http://", "")
104                    return res.split('/')[0]
105        return ""

 91    # Helper to extract the first domain-like token from identity text
 92    def _extract_domain(self, text: str) -> str:
 93        words = text.split()
 94        for w in words:
 95            w_clean = w.strip(".,;:()\"'!?")
 96            if "://" in w_clean:
 97                parts = w_clean.split("://")
 98                if len(parts) > 1:
 99                    return parts[1].split('/')[0]
100            if "." in w_clean and len(w_clean) > 4 and not w_clean.startswith("@"):
101                parts = w_clean.split(".")
102                last_part = parts[-1].split('/')[0].lower()
103                if last_part.isalnum() and 2 <= len(last_part) <= 6:
104                    res = w_clean.replace("https://", "").replace("http://", "")
105                    return res.split('/')[0]
106        return ""
